function Solution(list, count) {
  if (Math.abs(count) > list.length) {
    count =
      count > list.length
        ? count % list.length
        : (Math.abs(count) % list.length) * -1
  }

  return list.slice(count, list.length).concat(list.slice(0, count))
}
